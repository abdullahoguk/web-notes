d3.json("file.json",function(data){
    d3.select("body")
        .append("svg")
        .selectAll("rect")
        .data(data)
        .enter()
        .append("rect")
        .attr("y",function(d,i) {
            return i*24
        })
        .attr("width",function(d,i) {
            return 11*d.count
        })
        .attr("height", 20)
})
